this file contains tests for ``print_sorted()`` function in ``1-my_list`` module

>>> from importlib import import_module
>>> MyList = import_module('1-my_list').MyList
>>> print_sorted = MyList.print_sorted
>>> my_list = MyList()
>>> my_list.append(1)
>>> my_list.append(4)
>>> my_list.append(2)
>>> my_list.append(3)
>>> my_list.append(5)

# printing the original list
>>> print(my_list)
[1, 4, 2, 3, 5]

>>> my_list
[1, 4, 2, 3, 5]

# printing sorted list
>>> print_sorted(my_list)
[1, 2, 3, 4, 5]

>>> sorted_list = print_sorted(my_list)
>>> sorted_list
False

>>> my_list2 = MyList()
>>> my_list2.append(-4)
>>> my_list2.append(2)
>>> my_list2.append(0)
>>> my_list2.append(-6)
>>> print_sorted(my_list2)
[-6, -4, 0, 2]

>>> print_sorted([])
[]
